dim fs,d,n
set fs=Server.CreateObject("Scripting.FileSystemObject")
set d=fs.GetDrive("c:")

if d.IsReady=true then
    n = n & " drive is ready."
else
    n = n & " drive is not ready."
end if
Response.Write(n)
set d=nothing
set fs=nothing
